TCP

LengthCalcutator类

每当监听到了一个socket，都启动一个线程来执行相应代码，代码如下

package com.hwhjl.Socket;

import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import java.net.Socket;

public class LengthCalcutator extends Thread {
    private Socket socket;

    public LengthCalcutator(Socket socket) {
        this.socket = socket;
    }

    @Override
    public void run() {
        try {
            //获取socket输出流
            OutputStream os = socket.getOutputStream();
            //获取socket输出流
            InputStream is = socket.getInputStream();
            int ch = 0;
            byte[] buff = new byte[1024];
            //读取输入内容，存成字符串
            ch = is.read(buff);
            //将接受的byte数组转换为字符串
            String content = new String(buff, 0, ch);
            System.out.println(content);
            //往输入流里写入获取字符串里的长度，回发给客户端
            os.write(String.valueOf(content.length()).getBytes());
            //不要忘记关闭资源
            is.close();
            os.close();
            socket.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

服务端

package com.hwhjl.Socket;


import java.io.IOException;
import java.net.ServerSocket;
import java.net.Socket;

public class TCPServer {
    public static void main(String[] args) throws IOException {
        //创建Socket并绑定65000端口
        ServerSocket ss = new ServerSocket(65000);
        while(true){
            //监听65000端口，直到客户端返回连接信息后才返回
            Socket socket = ss.accept();
            //获取客户端的请求信息后，执行相关业务逻辑
            new LengthCalcutator(socket).start();
        }
    }
}

客户端

package com.hwhjl.Socket;

import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import java.net.Socket;

public class TCPClient {
    public static void main(String[] args) throws IOException {
        //创建socket，并且绑定本机ip和服务端端口
        Socket socket = new Socket("127.0.0.1",65000);
        //获取socket输入输出流
        OutputStream os = socket.getOutputStream();
        InputStream is = socket.getInputStream();
        //写入输出流
        os.write("Hello World!".getBytes());
        int ch = 0;
        byte[] buff = new byte[1024];
        //buff存数据，ch记录数组长度
        ch = is.read(buff);
        String content = new String(buff,0,ch);
        System.out.println(content);
        //不要忘记关闭资源
        is.close();
        os.close();
        socket.close();

    }

}

UDP

服务端

package com.hwhjl.Socket;

import java.io.IOException;
import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.SocketException;

public class UDPServer {
    public static void main(String[] args) throws IOException {
        //服务端接受接受客户端发送的数据包
        DatagramSocket socket = new DatagramSocket(65001);
        byte[] buff = new byte[100];
        DatagramPacket packet = new DatagramPacket(buff,buff.length);
        //封装到DatagramSocket对象中
        socket.receive(packet);

        byte[] data = packet.getData();
        //将二进制转换成字符串形式
        String content = new String(data,0,packet.getLength());
        System.out.println(content);
        //将发送给客户端的数据转换成二进制
        byte[] sendedContent = String.valueOf(content.length()).getBytes();
        //获取端口和地址
        DatagramPacket packetToClient = new DatagramPacket(sendedContent,sendedContent.length,
                packet.getAddress(),packet.getPort());
        //发送数据
        socket.send(packetToClient);
    }

}

客户端

package com.hwhjl.Socket;

import java.io.IOException;
import java.net.*;

public class UDPClient {
    public static void main(String[] args) throws IOException {
        //客户端发数据报给服务端
        DatagramSocket socket = new DatagramSocket();
        byte[] buff = "Hello World!".getBytes();
        //将IP地址封装成InetAddress对象
        InetAddress address = InetAddress.getByName("127.0.0.1");
        //将数据封装成packet，并写入ip和端口
        DatagramPacket packet = new DatagramPacket(buff,buff.length,address,65001);
        //发送数据
        socket.send(packet);

        //客户端接受服务端发过来的数据
        byte[] data = new byte[100];
        //创建packte对象存储接受的数据
        DatagramPacket receivedPacket = new DatagramPacket(data,data.length);
        //将数据封装到socket中
        socket.receive(receivedPacket);
        //打印到控制台
        String content = new String(receivedPacket.getData(),0,receivedPacket.getLength());
        System.out.println(content);
    }

}
